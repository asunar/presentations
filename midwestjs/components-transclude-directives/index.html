Digging further into Transclusion

  • transclude: true
  • transclude: 'element'
  • transclude: true + the 5th parameter to the linking function

transclude: true


  angular.module('midwestjs', [])
    .directive('jpAlertMsg', function() {
      return {
        restrict: 'E',
        transclude: true,
        template: '<div class='alert' ng-transclude></div>'
      };
    });
                                        

  <body ng-app="midwestjs">
    <jp-alert-msg>
      This is the message I want with a red background
    </jp-alert-msg>
  </body>
                                        

  <body ng-app="midwestjs">
    <div class='alert'>
      This is the message I want with a red background
    </div>
  </body>

transclude: true & 5th param to linking function


  angular.module('midwestjs', [])
    .directive('jpAlertMsg', function() {
      return {
        restrict: 'E',
        transclude: true,
        template: '<img src="expoint.svg"><div class="msg"></div>',
        link: function(scope, element, attrs, ctrls, transclude) {
          transclude(function(clone){
             element.find('.msg').append(clone);
          }); } }; }); 

    <jp-alert-msg>
      This is the message I want with a red background
    </jp-alert-msg>
                                        

    <jp-alert-msg>
        <img src="expoint.svg">
        <div class="msg">
            <span class="ng-scope">
              This is the message I want with a red background
            </span>
        </div>
    </jp-alert-msg>
