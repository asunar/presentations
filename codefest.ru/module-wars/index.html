RequireJS paths/shim


  requirejs.config({
      paths: {
          "jquery": "bower_components/jquery/jquery-2.1.0.min.js"
          "ember": "bower_components/ember/ember.js"
      },
      shim: {
          "ember": {
              deps: ["jquery"],
              exports: "Ember"
          }
      }
  });
