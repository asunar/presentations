Directives


  angular.module('emberconf', [])
    .directive('jpAlertMsg', function() {
      return {
        restrict: 'C',
        transclude: true,
        compile: function() {
          // stuff here
        },
        scope: {
          title: '&'
          name: '=',
          subTitle: '@',
        }
        template: '<div class='alert' ng-transclude></div>'
      };
    });
                                        

  <body ng-app="emberconf">
    <div class="jpAlertMsg" sub-title="test">Directive Example</div>
  </body>
