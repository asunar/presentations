function* inf(){
  var i = 0;
  while(true){
    yield ++i;
  }
}

function* take(generator, num){
  var i = 0;
  for(var val of generator){
    if(i++ === num){
      return;
    }
    yield val;
  }
}

for(var x of take(inf(), 3)){
  alert(x);
